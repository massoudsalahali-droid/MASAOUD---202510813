#include <iostream>
#include <string>
using namespace std;

int main() {
    
    string names[7] = {"bebo", "Amal", "Bakri", "Tayeb", "Omer", "Sami", "Hassan"};
    int n = 7;
    string temp;

    // Bubble Sort
    for(int i = 0; i < n-1; i++) {
        for(int j = 0; j < n-i-1; j++) {
            if(names[j] > names[j+1]) {
                temp = names[j];
                names[j] = names[j+1];
                names[j+1] = temp;
            }
        }
    }

    // Display
    cout << "Ascending Order:" << endl;
    for(int i = 0; i < n; i++) {
        cout << names[i] << endl;
    }

    return 0;
}
