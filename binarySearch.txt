#include <iostream>
#include <string>

using namespace std;

// 
void sortMembers(string names[], int ages[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (names[j] > names[j + 1]) {
                swap(names[j], names[j + 1]);
swap(ages[j], ages[j + 1]);            }
        }
    }
}

// 
int binarySearch(string names[], int n, string key) {
    int low = 0, high = n - 1;
    while (low <= high) {
        int mid = low + (high - low) / 2;
        if (names[mid] == key) return mid;
        if (names[mid] < key) low = mid + 1;
        else high = mid - 1;
    }
    return -1;
}

int main() {
    int n;
    cout << "How many family members? ";
    cin >> n;

    string names[n];
    int ages[n];

    for (int i = 0; i < n; i++) {
        cout << "Enter name of member " << i + 1 << ": ";
        cin >> names[i];
        cout << "Enter age of " << names[i] << ": ";
        cin >> ages[i];
    }

    sortMembers(names, ages, n); // 

    string key;
    cout << "\nEnter name to search: ";
    cin >> key;

    int index = binarySearch(names, n, key);

    if (index != -1) {
        cout << key << " was found and is aged " << ages[index] << endl;
    } else {
        cout << "Not found" << endl;
    }

    return 0;
}
